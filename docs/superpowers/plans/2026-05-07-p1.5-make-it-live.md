```python
from contextlib import contextmanager
from dataclasses import dataclass, field
from datetime import datetime, timezone
from uuid import uuid4

from isbe.facts.artifacts import TopicRun
from isbe.facts.db import make_session_factory


@dataclass
class _RunHandle:
    payload: dict = field(default_factory=dict)


@contextmanager
def topic_run(topic_id: str, flow_name: str):
    """Records a TopicRun row with start/finish/status. Yields a handle whose .payload dict
    is persisted to the row's payload column."""
    started = datetime.now(timezone.utc)
    handle = _RunHandle()
    status = "ok"
    err: str | None = None
    try:
        yield handle
    except Exception as e:  # noqa: BLE001
        status = "failed"
        err = f"{type(e).__name__}: {e}"
        raise
    finally:
        finished = datetime.now(timezone.utc)
        payload = dict(handle.payload)
        if err:
            payload["error"] = err
        Session = make_session_factory()
        with Session() as s:
            s.add(
                TopicRun(
                    id=uuid4(),
                    topic_id=topic_id,
                    flow_name=flow_name,
                    status=status,
                    started_at=started,
                    finished_at=finished,
                    payload=payload,
                )
            )
            s.commit()
```

Use `serve()` to bind multiple deployments under one process:

```python
# src/isbe/scheduler.py
from prefect import serve

from isbe.topics.nowcasting.collectors.arxiv import (
    arxiv_collector,
    arxiv_download_pdfs,
)
from isbe.topics.nowcasting.collectors.github import github_collector
from isbe.topics.nowcasting.digester import weekly_digester


def serve_nowcasting() -> None:
    """Long-running: serves 4 deployments with cron schedules."""
    arxiv_daily = arxiv_collector.to_deployment(
        name="nowcasting-arxiv-daily",
        cron="0 6 * * *",  # 06:00 daily
        parameters={"max_results": 50},
    )
    github_daily = github_collector.to_deployment(
        name="nowcasting-github-daily",
        cron="30 6 * * *",  # 06:30 daily
    )
    pdf_weekly = arxiv_download_pdfs.to_deployment(
        name="nowcasting-pdf-weekly",
        cron="0 7 * * 1",  # Monday 07:00 (before digest)
        parameters={"limit": 20},
    )
    weekly = weekly_digester.to_deployment(
        name="nowcasting-weekly-digester",
        cron="0 8 * * 1",  # Monday 08:00
    )
    serve(arxiv_daily, github_daily, pdf_weekly, weekly)


if __name__ == "__main__":
    serve_nowcasting()
```
